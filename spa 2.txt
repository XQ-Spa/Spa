        function renderServices() {
            const gridRecovery = document.getElementById('grid-recovery');
            const gridTherapy = document.getElementById('grid-therapy');
            const lang = translations[currentLang];
            
            gridRecovery.innerHTML = '';
            gridTherapy.innerHTML = '';

            serviceData.forEach(item => {
                const cardHtml = createCard(item, lang);
                if (item.cat === 'recovery') {
                    gridRecovery.innerHTML += cardHtml;
                } else if (item.cat === 'therapy') {
                    gridTherapy.innerHTML += cardHtml;
                }
            });
            
            // Kích ho?t l?i hi?u ?ng fade-in cho các card v?a du?c t?o m?i
            initFadeIn();
        }

        // --- H? th?ng b?t s? ki?n Scroll (Intersection Observer) ---
        const observerOptions = {
            root: null,
            rootMargin: '0px',
            threshold: 0.15 // Kích ho?t khi hi?n du?c 15% di?n tích
        };

        const fadeObserver = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('is-visible');
                    observer.unobserve(entry.target); // Ng?ng theo dői sau khi dă hi?n (ch? animation 1 l?n)
                }
            });
        }, observerOptions);

        function initFadeIn() {
            document.querySelectorAll('.fade-in-section').forEach(section => {
                fadeObserver.observe(section);
            });
        }

        async function getAIConsultation() {
            const userInput = document.getElementById('aiInput').value;
            const container = document.getElementById('aiResponseContainer');
            const textEl = document.getElementById('aiText');
            const btn = document.getElementById('aiBtn');

            if (!userInput.trim()) return;

            container.classList.remove('hidden');
            textEl.innerText = currentLang === 'vi' ? "AI dang phân tích..." : "AI is analyzing...";
            btn.disabled = true;

            const systemPrompt = `You are a senior wellness advisor at "XQ Spa Nha Trang". 
            Based on the client's symptoms, suggest the best treatment course from our menu.
            Language: Respond ONLY in ${currentLang}. Style: Warm, luxury, professional. Under 100 words.`;

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: userInput }] }],
                        systemInstruction: { parts: [{ text: systemPrompt }] }
                    })
                });
                const data = await response.json();
                textEl.innerText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Consultation error.";
            } catch {
                textEl.innerText = "Connection error.";
            } finally {
                btn.disabled = false;
            }
        }
